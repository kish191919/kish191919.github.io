[LAB] Anonymous functions

//--Lightweight anonymous functions....
scala> (x: Int) => x * x
res8: Int => Int = $Lambda$1199/2115607788@5d965dbd

//--This is a syntactic sugar for....
scala> new Function1[Int, Int] {
|   def apply(x: Int): Int = x * x
| }
res10: Int => Int = <function1>

//--Calling the anonymous function....
scala> val square = (x: Int) => x * x
square: Int => Int = $Lambda$1025/666312528@383790cf

scala> square(2)
res0: Int = 4

[LAB] Currying
------------------------------------------------------------------
//--Converting a function with multiple arguments into a function with a single argument that returns another function....
scala> def gen(f: Int => Int)(x: Int) =  f(x)
gen: (f: Int => Int)(x: Int)Int

scala> def identity(x: Int) = gen(i => i)(x)
identity: (x: Int)Int

scala> def square(x: Int) = gen(i => i * i)(x)
square: (x: Int)Int

scala> def cube(x: Int) = gen(i => i * i * i)(x)
cube: (x: Int)Int

scala> identity(2)
res2: Int = 2

scala> square(2)
res3: Int = 4

scala> cube(2)
res4: Int = 8


scala> def f100(f: Int => Int) = gen(f)(100)
f100: (f: Int => Int)Int

scala> def f50(f: Int => Int) = gen(f)(50)
f50: (f: Int => Int)Int

scala> f100(x => x * x)
res5: Int = 10000

scala> f50(x => x * x)
res7: Int = 2500

[LAB] Anonymous Parameters
------------------------------------------------------------------
//--Define function....
scala> def doWithOneAndTwo(f: (Int, Int) => Int) = f
doWithOneAndTwo: (f: (Int, Int) => Int)(Int, Int) => Int

//--Explicit type declaration....
scala> val call1 = doWithOneAndTwo((x: Int, y: Int) => x + y)
call1: (Int, Int) => Int = $Lambda$1075/1919747202@15ab0461

scala> call1(100, 100)
res8: Int = 200

//--The compiler expects 2 ints so x and y types are inferred....
scala> val call2 = doWithOneAndTwo((x, y) => x + y)
call2: (Int, Int) => Int = $Lambda$1076/1860013398@226f472

scala> call2(100, 100)
res9: Int = 200

//--Even more concise syntax by _(underscore)....
scala> val call3 = doWithOneAndTwo(_ + _)
call3: (Int, Int) => Int = $Lambda$1077/695653925@693afcd1

scala> call3(100, 100)
res10: Int = 200

[LAB] Returning multiple variables
------------------------------------------------------------------
//--Return multiple variables by Tuples....
scala> def swap(x: String, y: String) = (y, x)
swap: (x: String, y: String)(String, String)

scala> val (a, b) = swap("hello", "world")
a: String = world
b: String = hello

scala> val a, b = swap("hello", "world")
a: (String, String) = (world,hello)
b: (String, String) = (world,hello)

//--Return multiple variables by Array....
scala> def swap2(x: String, y: String) = Array(y, x)
swap2: (x: String, y: String)Array[String]

scala> val Array(a, b) = swap2("hi", "spark")
a: String = spark
b: String = hi

scala> val a, b = swap2("hi", "spark")
a: Array[String] = Array(spark, hi)
b: Array[String] = Array(spark, hi)

[LAB] High Order Functions
------------------------------------------------------------------
//--Methods that take as parameter functions....
scala> (1 to 4)
res17: scala.collection.immutable.Range.Inclusive = Range 1 to 4

scala> val list = (1 to 4).toList
list: List[Int] = List(1, 2, 3, 4)

scala> list.foreach(x => println(x))
1
2
3
4

scala> list.foreach(println)
1
2
3
4

scala> list.map(x => x + 2)
res18: List[Int] = List(3, 4, 5, 6)

scala> list.map(_ + 2)
res19: List[Int] = List(3, 4, 5, 6)

scala> list.filter(x => x % 2 == 1)
res20: List[Int] = List(1, 3)

scala> list.filter(_ % 2 == 1)
res21: List[Int] = List(1, 3)

scala> list.reduce((x, y) => x + y)
res22: Int = 10

scala> list.reduce(_ + _)
res23: Int = 10
