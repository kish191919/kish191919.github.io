(3)

spark 유저 생성

sudo user add spark -m -s /bin/bash

패스워드 변경

sudo passed spark

sudo 권한 부여

sudo visudo

편집모드에 아래 내용 추가

spark ALL=(ALL) NOPASSWD: ALL

spark 유저로 바꾸기

sudo su - spark

data 디렉토리 만들기

sudo mkdir /data

data의 오너와 그룹계정을 spark로 변경

sudo chown spark:spark /data

필요한 유틸리티 설치

sudo apt install -y wget unzip bzip2 net-tools

(4)

호스트 이름 설정

sudo hostnamectl set-hostname main-1
sudo hostnamectl set-hostname worker-1
sudo hostnamectl set-hostname worker-2
sudo hostnamectl set-hostname worker-3

호스트 이름 확인

hostname
or
cat /etc/hostname

프라이빗 IP 주소 확인

ifconfig

etc-host 파일 수정

sudo vi /etc/hosts

편집모드에 아래 내용 추가

00000000 main-1
00000000 worker-1
00000000 worker-2
00000000 worker-3

(5)

ssh 명령으로 worker-1에 접근

ssh worker-1

ssh 키 생성하기

ssh-keygen -t rsa -P ‘’ -f ~/.ssh/id_rsa

ssh 파일확인

ls -al ~/.ssh

private key 파일 내용확인

cat ~/.ssh/id_rsa

pubic key 파일 내용확인

cat ~/.ssh/id_rsa.pub

Authorized key 파일 생성하면서 퍼블릭 키 등록

cat ~/.ssh/id_rsa.pub >> ~/.ssh/authorized_keys

authorized keys 파일의 권한을 소유자만 읽고 편집할 수 있게 바꾸다

chmod 600 ~/.ssh/authorized_keys

(6)

로컬PC의 호스트 파일 수정

vi /etc/hosts

아래 퍼블릭 IP와 호스트이름을 추가

000.000.000.000 main
000.000.000.000 worker-1
000.000.000.000 worker-2
000.000.000.000 worker-3
